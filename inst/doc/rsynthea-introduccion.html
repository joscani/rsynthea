rsynthea: Generación de pacientes sintéticos para investigación

¿Qué es rsynthea?

rsynthea es un generador de historiales clínicos sintéticos para R. Simula cohortes de pacientes virtuales usando los 243 módulos de enfermedad del Generic Module Framework (GMF) de Synthea, produciendo datos que imitan la epidemiología de la población de Massachusetts (EEUU).

Casos de uso principales:

  • Prototipar pipelines de datos clínicos sin datos reales
  • Enseñanza y formación en salud computacional
  • Benchmarking de algoritmos de IA clínica
  • Estudios de simulación epidemiológica

Rendimiento (Linux x86, 20 pacientes):

Motor Tiempo vs Java Synthea
rsynthea C++ (serial) ~5s 2.7× más rápido
rsynthea C++ (6 cores) ~1.5s 9× más rápido

Uso básico

1. Generar una cohorte

Los módulos GMF se cargan y compilan automáticamente en la primera llamada y quedan en caché para el resto de la sesión:

library(rsynthea)

tbls <- generate_population(
  n        = 200,
  seed     = 42L,
  end_date = as.POSIXct("2023-01-01"),
  min_age  = 40L,
  max_age  = 65L
)

names(tbls)
#> [1] "patients"      "encounters"    "conditions"    "medications"
#> [4] "procedures"    "observations"  "allergies"     "immunizations"

Prevalencia de condiciones crónicas

prevalencia <- tbls$conditions |>
  filter(code %in% c("44054006", "714628002", "59621000", "195967001")) |>
  group_by(code, description) |>
  summarise(n_pacientes = n_distinct(patient_id), .groups = "drop") |>
  mutate(
    prevalencia = n_pacientes / nrow(tbls$patients) * 100,
    descripcion = case_when(
      code == "44054006" ~ "Diabetes tipo 2",
      code == "714628002" ~ "Prediabetes",
      code == "59621000" ~ "Hipertensión",
      code == "195967001" ~ "Asma",
      TRUE ~ description
    )
  )

ggplot(prevalencia, aes(x = reorder(descripcion, prevalencia), y = prevalencia)) +
  geom_col(fill = "#2196F3", alpha = 0.85) +
  geom_text(aes(label = sprintf("%.1f%%", prevalencia)), hjust = -0.1, size = 3.5) +
  coord_flip() +
  scale_y_continuous(limits = c(0, 80), labels = function(x) paste0(x, "%")) +
  labs(
    title    = "Prevalencia de condiciones crónicas",
    subtitle = "Cohorte sintética de 200 adultos (30-75 años)",
    x = NULL, y = "% pacientes"
  ) +
  theme_minimal(base_size = 12)

Comorbilidades del paciente diabético

dm_ids     <- tbls$conditions |> filter(code == "44054006") |> pull(patient_id) |> unique()
non_dm_ids <- tbls$patients$id[!tbls$patients$id %in% dm_ids]

cat(sprintf("Pacientes con diabetes tipo 2: %d / %d (%.1f%%)\n",
    length(dm_ids), nrow(tbls$patients),
    100 * length(dm_ids) / nrow(tbls$patients)))
#> Pacientes con diabetes tipo 2: 45 / 200 (22.5%)

comorb_codes <- c(
  "59621000"        = "Hipertensión",
  "82423001"        = "Dolor crónico",
  "368581000119106" = "Neuropatía",
  "195967001"       = "Asma"
)

comorb <- tbls$conditions |>
  filter(code %in% names(comorb_codes)) |>
  distinct(patient_id, code) |>
  mutate(
    es_diabetico = patient_id %in% dm_ids,
    condicion    = comorb_codes[code]
  ) |>
  group_by(condicion, es_diabetico) |>
  summarise(n = n(), .groups = "drop") |>
  mutate(
    total = ifelse(es_diabetico, length(dm_ids), length(non_dm_ids)),
    pct   = round(n / total * 100, 1),
    grupo = ifelse(es_diabetico, "Diabético", "No diabético")
  )

# Pivot wide para dumbbell (sin tidyr)
dm_pct   <- comorb |> filter(grupo == "Diabético")    |> select(condicion, Diabético      = pct)
nodm_pct <- comorb |> filter(grupo == "No diabético") |> select(condicion, `No diabético` = pct)
comorb_wide <- merge(dm_pct, nodm_pct, by = "condicion", all = TRUE) |>
  mutate(condicion = reorder(condicion, Diabético))

ggplot(comorb_wide) +
  geom_segment(
    aes(y = condicion, yend = condicion,
        x = `No diabético`, xend = Diabético),
    color = "grey70", linewidth = 2, na.rm = TRUE
  ) +
  geom_point(aes(y = condicion, x = `No diabético`),
             color = "#43A047", size = 5, na.rm = TRUE) +
  geom_point(aes(y = condicion, x = Diabético),
             color = "#E53935", size = 5) +
  geom_text(aes(y = condicion, x = `No diabético`,
                label = paste0(`No diabético`, "%")),
            color = "#43A047", vjust = -1.2, size = 3.5, na.rm = TRUE) +
  geom_text(aes(y = condicion, x = Diabético,
                label = paste0(Diabético, "%")),
            color = "#E53935", vjust = -1.2, size = 3.5) +
  scale_x_continuous(labels = function(x) paste0(x, "%"), limits = c(0, 105)) +
  labs(
    title    = "Prevalencia de comorbilidades por grupo",
    subtitle = "Punto rojo: % diabéticos con la condición  ·  Punto verde: % no diabéticos",
    x = "% de pacientes del grupo con la condición", y = NULL
  ) +
  theme_minimal(base_size = 12) +
  theme(panel.grid.major.y = element_blank(),
        panel.grid.minor   = element_blank())

Ejemplo avanzado: seguimiento longitudinal

rsynthea genera historiales completos desde el nacimiento. Esto permite análisis longitudinales reales, como la evolución de la HbA1c en pacientes diabéticos a lo largo del tiempo (código LOINC 4548-4).

ha1c <- tbls$observations |>
  filter(code == "4548-4") |>
  inner_join(
    tbls$conditions |> filter(code == "44054006") |> distinct(patient_id),
    by = "patient_id"
  ) |>
  mutate(
    anyo       = as.integer(format(time, "%Y")),
    ha1c_valor = as.numeric(value)
  ) |>
  filter(!is.na(ha1c_valor), ha1c_valor > 4, ha1c_valor < 15) |>
  group_by(anyo) |>
  summarise(
    media = mean(ha1c_valor),
    p25   = quantile(ha1c_valor, 0.25),
    p75   = quantile(ha1c_valor, 0.75),
    n     = n(),
    .groups = "drop"
  )

cat(sprintf("Observaciones HbA1c en diabéticos: %d  |  Años: %d–%d\n",
    sum(ha1c$n), min(ha1c$anyo), max(ha1c$anyo)))
#> Observaciones HbA1c en diabéticos: 1371  |  Años: 1966–2022

ggplot(ha1c, aes(x = anyo)) +
  geom_ribbon(aes(ymin = p25, ymax = p75), alpha = 0.2, fill = "#E53935") +
  geom_line(aes(y = media), color = "#E53935", linewidth = 1) +
  geom_hline(yintercept = 6.5, linetype = "dashed", color = "grey40") +
  annotate("text", x = min(ha1c$anyo), y = 6.65,
           label = "Umbral diagnóstico (6.5%)", hjust = 0, size = 3) +
  scale_x_continuous(breaks = scales::breaks_pretty(n = 8)) +
  labs(
    title    = "HbA1c media en pacientes diabéticos a lo largo del tiempo",
    subtitle = "Banda: IQR (p25–p75)",
    x = NULL, y = "HbA1c (%)"
  ) +
  theme_minimal(base_size = 12)

Notas sobre los datos generados

Los datos producidos por rsynthea son sintéticos y no corresponden a personas reales. Sin embargo, los módulos GMF están calibrados para reproducir:

  • Prevalencias de enfermedades crónicas de Massachusetts (EEUU)
  • Distribuciones de edad, género y etnia
  • Progresión natural de enfermedades (diabetes, hipertensión, EPOC…)
  • Patrones de encuentros médicos y prescripción

Para generar poblaciones representativas de otras regiones, considerar ajustar las probabilidades de los módulos base.
